# Mobius

# Typical price is based on a Stochastic Average

input colorCandles = yes;
def x = barNumber();

def TYP = (highest(h, n) + lowest(l, n) + c) / 3;
def dn_rev = if c crosses below TYP
             then l
             else if c > TYP
                  then nan
                  else if l < dn_rev[1]
                       then l
                       else dn_rev[1];
def dn_x = if l == dn_rev
           then x-1
           else nan;

            then h[1]
            else h_rev[1];

               then highestAll(if isNaN(close[-1])

     ArrowUp.SetLineWeight(2);
     ArrowUp.SetDefaultColor(color.green);

             then h
             else if c < TYP
                  else if h > up_rev[1]
                       then h
                       else up_rev[1];
def up_x = if h == up_rev

# Custom Aggregations
# V01.01.2017
# Chart aggregation MUST be lower then custom aggregation and custom aggregation MUST be divisible without remainder by chart aggregation.
input AggPeriodMin = 8;
def ChartAggMin = getAggregationPeriod() / 1000 / 60;
def multiplier = AggPeriodMin / ChartAggMin; 
def o = if x % multiplier == 0
        then open
        else o[1];
def h = if x % multiplier == 0
        then high
        else max(high, h[1]);
def l = if x % multiplier == 0
        then low
        else min(low, l[1]);
