# RSI in Laguerre Time Scaled to Price 

# Mobius

script Scale { 

 input c = close; 

 input Min = 0; 

 input Max = 1; 

 def hh = HighestAll(c); 

 def ll = LowestAll(c); 

 plot Range = (((Max - Min) * (c - ll)) /  (hh - ll)) + Min; 

}

L3 = -gamma * L2 + L2[1] + gamma * L3[1];
